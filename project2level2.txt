// Extend Level 1 with sorting
// Example: Selection Sort Patients by Age, Insertion Sort Doctors by Specialty
// Merge Sort Appointments (define Appointment class with time)

#include <iostream>
#include <list>
#include <string>
#include <algorithm>
using namespace std;

// (Reuse Person, Patient, Doctor, Nurse, Hospital from Level 1)

class Appointment {
    string patientName;
    int time; // simple integer for time
public:
    Appointment(string p, int t):patientName(p),time(t){}
    int getTime() const { return time; }
    void display() { cout<<"Appointment: "<<patientName<<" at "<<time<<endl; }
};

// Sorting functions
void sortPatientsByAge(list<Person*>& people) {
    people.sort([](Person* a, Person* b){
        Patient* pa=dynamic_cast<Patient*>(a);
        Patient* pb=dynamic_cast<Patient*>(b);
        if(pa && pb) return pa->getAge()<pb->getAge();
        return false;
    });
}

void sortDoctorsBySpecialty(list<Person*>& people) {
    people.sort([](Person* a, Person* b){
        Doctor* da=dynamic_cast<Doctor*>(a);
        Doctor* db=dynamic_cast<Doctor*>(b);
        if(da && db) return da->getSpecialty()<db->getSpecialty();
        return false;
    });
}

void mergeSortAppointments(vector<Appointment>& arr) {
    if(arr.size()<=1) return;
    int mid=arr.size()/2;
    vector<Appointment> left(arr.begin(),arr.begin()+mid);
    vector<Appointment> right(arr.begin()+mid,arr.end());
    mergeSortAppointments(left);
    mergeSortAppointments(right);
    arr.clear();
    auto i=left.begin(), j=right.begin();
    while(i!=left.end() && j!=right.end()) {
        if(i->getTime()<j->getTime()) arr.push_back(*i++);
        else arr.push_back(*j++);
    }
    while(i!=left.end()) arr.push_back(*i++);
    while(j!=right.end()) arr.push_back(*j++);
}
